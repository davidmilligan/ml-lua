-- draws a clock

function event.seconds_clock()
    if menu.visible == false and lv.enabled == false then
        local current_time = dryos.date
        local ampm = "AM"
        local bg = display.pixel(339,410)
        if current_time.hour > 11 then ampm = "PM" end
        local time_str = string.format("%d:%02d:%02d %s",(current_time.hour + 11) % 12 + 1,current_time.min,current_time.sec,ampm)
        display.print(time_str,340,410,FONT.LARGE,COLOR.WHITE,bg)
    end
end
