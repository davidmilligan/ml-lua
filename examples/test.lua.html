booltostring = function(value)
	if value then return "true" else return "false" end
end

testmenu = menu.new{
	name = "Test Param for script",
    choices = { "choice1", "choice2", "choice3" },
	depends_on = DEPENDS_ON.LIVEVIEW,
	info = function() return "click it baby!" end,
}

runmenu = menu.new{
    name = "Run Test Script",
    help = "Run the test script.",
    submenu =
    {
        {
            name = "Run",
            help = "Run this script.",
            icon_type = ICON_TYPE.ACTION,
            update = "",
        },
        {
            name = "param1",
            help = "help for param1",
            min = 0,
            max = 100,
            warning = function() return "this param doesn't work right :P" end,
        },
        {
            name = "param2",
            help = "help for param2",
            min = 0,
            max = 10,
            value = 5,
            depends_on = DEPENDS_ON.LIVEVIEW,
            info = function() return "click it baby!" end,
        },
        {
            name = "dec test",
            min = 0,
            max = 10000,
            unit = UNIT.DEC,
        },
        {
            name = "choices test",
            choices = { "choice1", "choice2", "choice3" },
        }
    },
    update = function() return runmenu.submenu["choices test"].value end,
}

runmenu.submenu["Run"].select = function()
	console.show()
	print("choices="..runmenu.submenu["choices test"].value)
	print("shutter = "..camera.shutter)
	print("aperture = "..camera.aperture)
	print("iso = "..camera.iso)
	print("ec = "..camera.ec)
	print("model = "..camera.model)
	print("lens = "..lens.name)
	print("lv = "..lv.enabled)

	print("script run finished!")
end
