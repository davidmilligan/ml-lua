require("logger")

--[[---------------------------------------------------------------------------
Helper functions
]]
function printf(s,...)
    print(s:format(...))
end

function log_table(t)
    local log = logger("LUATEST.LOG")
    log:write(string.format("%s = ", t))
    local s,e = xpcall(function()
        log:serialize(_G[t])
    end, debug.traceback)
    if s == false then
        log:write(e)
    end
    log:close()
end

--[[---------------------------------------------------------------------------
Verify apex fields are 'reversible'
]]
function test.apex()
   display.clear()
   a = camera.shutter.apex
   camera.shutter.apex = 4
   b = camera.shutter.apex
   camera.shutter.apex = a
   c = camera.shutter.apex
   i = a..","..b..","..c
   display.notify_box (i,5000)
end

end
